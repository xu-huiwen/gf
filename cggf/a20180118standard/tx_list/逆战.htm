逆战demo

逆战与逆战页游活动专题导航原生js导航插件

1、调用方式：（非常简单），只需要在页面的最后引入公共导航js及其调用方式；

<script src="//ossweb-img.qq.com/images/js/nz/nact.min.js"></script>
<script>
new actNav({
actChannel:"15217",//频道id(一般不需要变)
actPos:"18700",//LOGO广告位id(一般不需要变)
actPosNormal:"18702",//背景图广告位id(一般不需要变)
actPosHover:"18703",//聚焦状态图片广告位id(一般不需要变)
sUrl:"navInfoListnz",//对应脚本库名字，逆战的对应值："navInfoListnz",逆战页游的对应值："navInfoListnzw"
colorNormal:"#fff",//常规状态文字色值,用于换肤自定义文字颜色
colorHover:"#0c0c0c"//聚焦状态文字色值,用于换肤自定义文字颜色
});
</script>

2、参数说明：

actChannel:"15217",//频道id(一般不需要变)对应项目广告系统的id；（外包一般不用管，直接复制过来调用就可以了）

actPosNormal:"18702",//背景图广告位id(一般不需要变)这个参数是对应一张带纹理的循环的背景图片，对应的广告位id（外包一般不用管，直接复制过来调用就可以了）；

actPosHover:"18703",//聚焦状态图片广告位id(一般不需要变)这个参数是对应一张横向循环的背景图片，对应的广告位id（外包一般不用管，直接复制过来调用就可以了）；

sUrl:"navInfoListnz",//对应脚本库名字，逆战的对应值："navInfoListnz",逆战页游的对应值："navInfoListnzw"这个参数对应的是后台开发提供的脚本数据库的名字，逆战的对应值这里写 navInfoListnz，而逆战页游调用该控件的时候这里要改为navInfoListnzw；

colorNormal:"#fff",//常规状态文字色值,用于换肤自定义文字颜色这个参数是常规状态每项文字的颜色；

colorHover:"#0c0c0c"//聚焦状态文字色值,用于换肤自定义文字颜色这个参数是鼠标hover到标签上的文字色值，这里日期的值一般保持不变，只改变主题文字的色值；
